
# block_info.json
# "0x0" 表示获取块号为 0 的区块，即第一个区块。
# true 表示同时获取区块中的交易详情。
curl localhost:8545 -X POST --data '{"jsonrpc":"2.0","method":"eth_getBlockByNumber","params":["0x4", true],"id":1}' | jq '.' > trace/t04_c_transfer_from_a_b_200/block_info.json


# tx_debug_trace.json
# 根据交易Id获取交易的tx trace
curl localhost:8545 \
   -X POST \
   -H "Content-Type: application/json" \
   --data '{"method":"debug_traceTransaction","params":["0xc2f4c1259a3f073d11ac79045d06e1a28dfa9d28e29722df6ffe33f36fb8efef"], "id":1,"jsonrpc":"2.0"}' | jq '.'  > trace/t04_c_transfer_from_a_b_200/tx_debug_trace.json

# tx_info.json
curl localhost:8545 -X POST --data '{"jsonrpc":"2.0","method":"eth_getTransactionByHash","params":["0xc2f4c1259a3f073d11ac79045d06e1a28dfa9d28e29722df6ffe33f36fb8efef"],"id":1}' | jq '.'  > trace/t04_c_transfer_from_a_b_200/tx_info.json
