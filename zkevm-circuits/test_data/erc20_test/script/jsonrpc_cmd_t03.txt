
# block_info.json
# "0x0" 表示获取块号为 0 的区块，即第一个区块。
# true 表示同时获取区块中的交易详情。
curl localhost:8545 -X POST --data '{"jsonrpc":"2.0","method":"eth_getBlockByNumber","params":["0x3", true],"id":1}' | jq '.' > trace/t03_a_approve_c_200/block_info.json


# tx_debug_trace.json
# 根据交易Id获取交易的tx trace
curl localhost:8545 \
   -X POST \
   -H "Content-Type: application/json" \
   --data '{"method":"debug_traceTransaction","params":["0x954b953137f225b0d4579a0f6f1ca1cea43640ab2ffc62e0aa824bb4e318b9e6"], "id":1,"jsonrpc":"2.0"}' | jq '.'  > trace/t03_a_approve_c_200/tx_debug_trace.json
